methods:
    getType(ctx): "
        var data = ctx.context.result;
        var type;
        var desc = '';
        var number = 1;
        for(let a in data){
            type = data[a].type;
            desc = desc + number + '. ' + type + '\\n';

            number += 1;
        }

        ctx.context.type = desc;

        return ctx;
    "
    getColor(ctx): "
        var data = ctx.context.result;
        var color;
        var desc = '';
        var number = 1;
        for(let b in data){
            type = data[b].color;
            desc = desc + number + '. ' + color + '\\n';

            number += 1;
        }

        ctx.context.color = desc;

        return ctx;
    "
